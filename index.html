  <script>

    // Simple interactive dot grid

    const grid = document.getElementById('dot-grid');

    const hero = document.querySelector('.hero');


    const cols = 24;   // adjust density horizontally

    const rows = 12;   // adjust density vertically

    const dots = [];


    function createDots() {

      const { width, height } = hero.getBoundingClientRect();

      grid.innerHTML = "";

      dots.length = 0;


      for (let r = 0; r < rows; r++) {

        for (let c = 0; c < cols; c++) {

          const dot = document.createElement('div');

          dot.className = 'dot';


          const x = (c + 0.5) * (width / cols);

          const y = (r + 0.5) * (height / rows);


          dot.style.left = x + 'px';

          dot.style.top = y + 'px';


          dot.dataset.x = x;

          dot.dataset.y = y;


          grid.appendChild(dot);

          dots.push(dot);

        }

      }

    }


    function handleMove(e) {

      const rect = hero.getBoundingClientRect();

      const mouseX = e.clientX - rect.left;

      const mouseY = e.clientY - rect.top;


      const maxDist = Math.min(rect.width, rect.height) * 0.35;


      for (const dot of dots) {

        const dx = mouseX - dot.dataset.x;

        const dy = mouseY - dot.dataset.y;

        const dist = Math.sqrt(dx*dx + dy*dy);


        let scale = 1;


        if (dist < maxDist) {

          const influence = 1 - dist / maxDist; // 0–1

          scale = 1 + influence * 0.9;          // grow up to ~1.9

        }


        dot.style.transform = `translate(-50%, -50%) scale(${scale})`;

      }

    }


    window.addEventListener('resize', createDots);

    hero.addEventListener('mousemove', handleMove);


    createDots();

  </script>
